# Define a function to perform addition
function add(numbers):
    return sum(numbers)

# Define a function to perform subtraction
function subtract(a, b):
    return a - b

# Define a function to perform multiplication
function multiply(numbers):
    result = 1
    for number in numbers:
        result = result * number
    return result

# Define a function to perform division
function divide(a, b):
    if b == 0:
        return "Error: Division by zero"
    return a / b

# Define a function to calculate the average
function average(numbers):
    return sum(numbers) / length(numbers)
